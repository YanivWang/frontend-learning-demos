条件类型与 infer

type IsString<T> = T extends string ? true : false;

type MyReturnType<T> =
  T extends (...args: any[]) => infer R ? R : never;

type MyAwaited<T> =
  T extends Promise<infer R> ? MyAwaited<R> : T;

type A = ReturnType<() => number>;  // number
type B = MyAwaited<Promise<Promise<string>>>;  // string
