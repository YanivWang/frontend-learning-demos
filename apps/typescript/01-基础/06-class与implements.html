class、implements 与 abstract

知识点要点

  • implements 检查 class 是否满足接口形状，但不会改变运行时行为。
  • public / private / protected 是 TS 类型层面的访问约束。
  • abstract class 可放公共实现，子类必须实现抽象成员。
  • 面试要能说清 interface、abstract class、普通 class 的边界。

interface Repository<T> {
  findById(id: number): T | undefined;
}

abstract class BaseService {
  protected prefix = "service";
  abstract label(): string;
}

class UserService extends BaseService implements Repository<User> {
  private users: User[] = [];
  findById(id: number) { return this.users.find((u) => u.id === id); }
  label() { return this.prefix + ":user"; }
}
