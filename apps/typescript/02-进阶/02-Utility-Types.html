Utility Types

知识点要点

  • Utility Types 是 TS 内置类型工具，面试常要求说明用途并手写简化版。
  • Partial 把属性变可选，Required 把属性变必填，Readonly 把属性变只读。
  • Pick / Omit 用于从对象类型中选择或排除字段。
  • Record 常用于字典表、枚举映射和权限配置。

interface User {
  id: number;
  name: string;
  email: string;
  role: "admin" | "guest";
}

type UserPatch = Partial<Pick<User, "name" | "email">>;
type PublicUser = Omit<User, "email">;
type RoleLabels = Record<User["role"], string>;

const roleLabels: RoleLabels = { admin: "管理员", guest: "访客" };
